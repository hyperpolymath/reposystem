= Ecosystem Integration Layer
:toc: left
:toclevels: 3
:icons: font

[.lead]
Integration design connecting git-dispatcher with scaffoldia, stateful-artefacts-for-gitforges, reposystem, git-hud, git-seo, and gitbot-fleet.

== Overview

The integration layer enables:

* **Scaffoldia** to be invoked as a git-dispatcher operation
* **Stateful-artefacts (Gnosis)** to auto-update from git-seo and git-hud
* **Reposystem** to track scaffolded repos automatically
* **Git-HUD** to display integration status
* **Cross-tool data flow** via STATE.scm updates

== New Operation Types for git-dispatcher

=== CreateScaffold Operation

Create a new repository from a scaffoldia template.

[source,rescript]
----
type operationType =
  | ...
  | CreateScaffold({
      template: string,        // "rust-library", "elixir-service", etc.
      destination: string,     // "/path/to/new-repo"
      repoName: string,        // "my-new-library"
      metadata: Js.Dict.t<string>,  // {"author": "...", "license": "PMPL-1.0"}
    })
----

**Execution flow:**

[source,rescript]
----
let executeCreateScaffold = async (
  template: string,
  destination: string,
  repoName: string,
  metadata: Js.Dict.t<string>,
): Plan.opResult => {
  // 1. Call scaffoldia CLI
  let scaffoldiaCmd = `scaffoldia generate \
    --template ${template} \
    --output ${destination} \
    --name ${repoName}`

  // Add metadata as arguments
  metadata->Js.Dict.entries->Array.forEach(((key, value)) => {
    scaffoldiaCmd = scaffoldiaCmd ++ ` --${key} ${value}`
  })

  // 2. Execute scaffoldia
  let result = await Deno.Command.output({
    cmd: "bash",
    args: ["-c", scaffoldiaCmd],
  })

  if result.success {
    // 3. Register in reposystem
    await registerNewRepo(destination, repoName)

    // 4. Initialize gnosis auto-rendering
    await initializeGnosis(destination)

    // 5. Return success
    return {
      opId: generateId(),
      status: Completed,
      output: Some(`Scaffolded ${repoName} at ${destination}`),
      ...
    }
  } else {
    return {
      opId: generateId(),
      status: Failed({error: "Scaffolding failed"}),
      ...
    }
  }
}
----

=== UpdateMetadataFromSeo Operation

Update STATE.scm with git-seo analysis results.

[source,rescript]
----
type operationType =
  | ...
  | UpdateMetadataFromSeo({
      repoPath: string,
      runAnalysis: bool,  // If false, read existing report
    })
----

**Execution flow:**

[source,rescript]
----
let executeUpdateMetadataFromSeo = async (
  repoPath: string,
  runAnalysis: bool,
): Plan.opResult => {
  // 1. Run or load git-seo analysis
  let report = if runAnalysis {
    await runGitSeoAnalysis(repoPath)
  } else {
    await loadGitSeoReport(repoPath)
  }

  // 2. Extract values
  let score = report.scores.total
  let recommendations = report.recommendations

  // 3. Update STATE.scm
  let stateFile = repoPath ++ "/.machine_readable/STATE.scm"
  await updateSCMValue(stateFile, "seo-score", score->Float.toString)
  await updateSCMValue(stateFile, "seo-recommendations", recommendations->Array.length->Int.toString)

  // 4. Trigger gnosis re-render
  await renderGnosisTemplates(repoPath)

  // 5. Return result
  return {
    opId: generateId(),
    status: Completed,
    output: Some(`Updated SEO score: ${score->Float.toString}/100`),
    ...
  }
}
----

=== RenderDocumentation Operation

Trigger gnosis to re-render all templates.

[source,rescript]
----
type operationType =
  | ...
  | RenderDocumentation({
      repoPath: string,
      templates: array<string>,  // Specific templates or [] for all
    })
----

**Execution flow:**

[source,rescript]
----
let executeRenderDocumentation = async (
  repoPath: string,
  templates: array<string>,
): Plan.opResult => {
  // 1. Find all .template.md files
  let templateFiles = if Array.length(templates) == 0 {
    await findAllTemplates(repoPath)
  } else {
    templates
  }

  // 2. Call gnosis for each template
  let results = await Promise.all(
    templateFiles->Array.map(async templateFile => {
      let outputFile = templateFile->String.replace(".template.", ".")
      let cmd = `gnosis ${templateFile} ${outputFile}`
      await executeCommand(cmd)
    })
  )

  // 3. Check if any failed
  let failures = results->Array.keep(r => !r.success)

  if Array.length(failures) == 0 {
    return {
      opId: generateId(),
      status: Completed,
      output: Some(`Rendered ${Array.length(results)} templates`),
      ...
    }
  } else {
    return {
      opId: generateId(),
      status: Failed({error: `${Array.length(failures)} templates failed`}),
      ...
    }
  }
}
----

=== RegisterInReposystem Operation

Register a repository in reposystem graph.

[source,rescript]
----
type operationType =
  | ...
  | RegisterInReposystem({
      repoPath: string,
      repoName: string,
      aspects: array<string>,  // ["security", "reliability", ...]
      group: option<string>,
    })
----

**Execution flow:**

[source,rescript]
----
let executeRegisterInReposystem = async (
  repoPath: string,
  repoName: string,
  aspects: array<string>,
  group: option<string>,
): Plan.opResult => {
  // 1. Add repo to reposystem
  await runCommand(`reposystem scan ${repoPath}`)

  // 2. Tag with aspects
  await Promise.all(
    aspects->Array.map(async aspect => {
      await runCommand(`reposystem aspect tag ${repoName} ${aspect}`)
    })
  )

  // 3. Add to group if specified
  switch group {
  | Some(groupName) =>
    await runCommand(`reposystem group add ${groupName} ${repoName}`)
  | None => ()
  }

  // 4. Export updated graph
  await runCommand(`reposystem export --format json > graph.json`)

  return {
    opId: generateId(),
    status: Completed,
    output: Some(`Registered ${repoName} in reposystem`),
    ...
  }
}
----

== Integration Workflows

=== Workflow 1: New Repository Creation

**Trigger:** User runs `reposystem plan create-repo`

**Steps:**

[source,text]
----
1. User: reposystem plan create --template rust-library --name my-lib
   └─> Creates plan with CreateScaffold operation

2. User: git-dispatcher execute plan-123
   └─> Loads plan, validates

3. git-dispatcher → scaffoldia:
   └─> scaffoldia generate --template rust-library --output ~/repos/my-lib

4. scaffoldia creates:
   ├── Cargo.toml
   ├── src/lib.rs
   ├── .machine_readable/STATE.scm
   ├── README.template.adoc  ← Has (:completion), (:phase) placeholders
   └── .github/workflows/gnosis-render.yml

5. git-dispatcher → reposystem:
   └─> reposystem scan ~/repos/my-lib
   └─> Adds to graph

6. git-dispatcher → gnosis:
   └─> gnosis README.template.adoc README.adoc
   └─> Renders with current STATE.scm values

7. Result: New repo with auto-updating docs, tracked in reposystem
----

=== Workflow 2: SEO Score Auto-Update

**Trigger:** git-seo analysis completes (CI/CD or manual)

**Steps:**

[source,text]
----
1. CI runs: git-seo analyze --json > seo-report.json

2. CI triggers: git-dispatcher execute seo-update-plan
   └─> Plan contains: UpdateMetadataFromSeo operation

3. git-dispatcher → git-seo:
   └─> Reads seo-report.json
   └─> Extracts score: 87/100

4. git-dispatcher → STATE.scm:
   └─> Updates (seo-score . "87")
   └─> Updates (seo-last-updated . "2026-02-07")

5. git-dispatcher → gnosis:
   └─> gnosis render-all
   └─> Re-renders README with new score

6. Result: README now shows "SEO Score: 87/100" automatically
----

=== Workflow 3: Ecosystem-Wide Updates

**Trigger:** Monthly stats update cron job

**Steps:**

[source,text]
----
1. Cron: reposystem scan --all
   └─> Re-scans all 570+ repos
   └─> Updates graph with current state

2. Cron: for each repo:
   a. git-seo analyze --json > $repo/seo-report.json
   b. git-dispatcher execute update-metadata-$repo
      └─> Updates STATE.scm with SEO score

3. Cron: git-dispatcher execute render-all-docs
   └─> Triggers gnosis on all repos with templates

4. Result: All 570+ READMEs updated with current stats
----

== Configuration Files

=== .integration.toml

New config file for integration settings:

[source,toml]
----
# .integration.toml - Ecosystem integration config

[scaffoldia]
enabled = true
templates_path = "~/.scaffoldia/templates"

[gnosis]
enabled = true
auto_render = true

=== GitHub Actions Workflow

[source,yaml]
----
# .github/workflows/ecosystem-integration.yml
name: Ecosystem Integration

on:
  push:
    branches: [main]
  schedule:
    - cron: '0 0 * * 0'  # Weekly

jobs:
  update-and-render:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Run git-seo analysis
        run: |
          git-seo analyze --json > seo-report.json
          echo "SEO_SCORE=$(jq '.scores.total' seo-report.json)" >> $GITHUB_ENV

      - name: Update STATE.scm
        run: |
          sed -i "s/(seo-score . \"[0-9]*\")/(seo-score . \"$SEO_SCORE\")/" \
            .machine_readable/STATE.scm
          sed -i "s/(seo-last-updated . \".*\")/(seo-last-updated . \"$(date -I)\")/" \
            .machine_readable/STATE.scm

      - name: Render templates with gnosis
        run: |
          gnosis README.template.adoc README.adoc
          gnosis PROFILE.template.md PROFILE.md

      - name: Commit updates
        run: |
          git config user.name "Gnosis Bot"
          git config user.email "bot@hyperpolymath.org"
          git add .
          git commit -m "docs: auto-update from SEO analysis (score: $SEO_SCORE)"
          git push
----
